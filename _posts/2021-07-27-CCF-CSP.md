# 2021-04-2 邻域均值
题目要求输出图像中处于较暗区域的像素总数，很容易可以想到通过遍历整个矩阵，对每个元素求其邻域内所有元素的平均值来判断其是否处于较暗区域。
问题自然就到了如何求一个元素的邻域内所有元素的平均值。
一开始直接暴力解，套了4个for循环，结果自然是对的，但是时间复杂度显然太大，所以运行超时只有70分。
优化的想法是，首先遍历矩阵中所有元素的二重循环是不可避免的，因为要判断所有像素是不是处于较暗位置。但是如果对每一个元素，计算其邻域中所有元素之和还使用循环的话，那复杂度显然就太高了。所以现在想要做的是，对于每一个给定的元素，我能够直接给出其邻域内所有元素之和。为了做到这一点，重新定义了一个数组 S 。
~~（优化了，超时是没超了，但是显示错误只有70分，裂开，不想改了）~~
破案了，浮点数用 double 类型即可。
```cpp
#include <iostream>
using namespace std;

/*
    该函数用来在输入 A 数组的同时建立 S 数组。
    其中 S[i][j] 表示 A[0][0] 到 A[i][j] 之间
    矩形区域内所有元素之和。
*/
void cals(int i, int j, int **A, int **S){
    int sum = 0;
    sum = sum + A[i][j];
    if(i-1 >= 0)
        sum = sum + S[i-1][j];
    if(j-1 >= 0)
        sum = sum + S[i][j-1];
    if(i-1>=0 && j-1>=0)
        sum = sum - S[i-1][j-1];
    S[i][j] = sum;
}

// 用于计算 Aij 邻域中所有元素的平均值
double neighber(int i, int j, int r, int n, int **A, int **S){
    // 先计算横纵坐标的上下界
    int top = max(i-r, 0);
    int bottom = min(r+i, n-1);
    int left = max(j-r, 0);
    int right = min(r+j, n-1);

    int scale = (bottom-top+1)*(right-left+1); // 邻域规模（邻域中元素个数）

    double sum = 0;
    sum = sum + S[bottom][right];

    /*
        相当于求部分面积：
        大面积减去小面积再加上多减的面积（如果有的话）
    */
    if(left-1>=0)
        sum = sum - S[bottom][left-1];
    if(top-1>=0)
        sum = sum - S[top-1][right];
    if(top-1>=0 && left-1>=0)
        sum = sum + S[top-1][left-1];

    return (sum/scale);
}

int main(){
    int n, L, r, t;
    double ave=0; // 邻域平均值
    int num=0; // 处于较暗区域的像素数
    int i, j;
    cin >> n;
    cin >> L;
    cin >> r;
    cin >> t;

    // 创建二维数组A
    int **A;
    A = new int*[n];
    for(i = 0; i < n; i++)
        A[i] = new int[n];

    int **S;
    S = new int*[n];
    for(i = 0; i < n; i++)
        S[i] = new int[n];

    for(i = 0; i < n; i++){
        for(j = 0; j < n; j++){
            cin >> A[i][j];
            cals(i, j, A, S);
        }
    }

    // 遍历A，为每一个元素计算其邻域中所有元素的平均值
    for(i = 0; i < n; i++){
        for(j = 0; j < n; j++){
            ave = neighber(i, j, r, n, A, S);
            if(ave <= t)
                num++;
        }
    }

    cout << num << endl;
    return 0;
}
```

# 2020-12-2 期末预测之最佳阈值
求解问题很简单，只需要将每个y[i]作为阈值来计算预测正确的次数，再取最大次数对应的最大y[i]即可。
所以很容易能够想到用两个循环来做：

```cpp
#include <iostream>
using namespace std;

int threshold(int *y, int *result, int m){
    int th = 0; // 最终阈值
    int sum = 0; // 预测正确次数
    int maxsum = 0; // 最大正确次数
    int ymax = 0; // 最大y[i]
    for(int i = 0; i < m; i++){
        sum = 0;
        // y[i]为阈值
        for(int j = 0; j < m; j++){
            // 循环计算正确次数
            if(y[j]>=y[i] && result[j]==1)
                sum++;
            else if(y[j]<y[i] && result[j]==0)
                sum++;
        }
        if(sum > maxsum){
            maxsum = sum;
            ymax = y[i];
        }
        else if(sum==maxsum && y[i]>ymax)
            ymax = y[i];
    }
    th = ymax;
    return th;
}
int main(){
    int m, i, th;
    cin >> m;
    int *y = new int[m];
    int *result = new int[m];
    for(i = 0; i < m; i++){
        cin >> y[i];
        cin >> result[i];
    }
    th = threshold(y, result, m);
    cout << th << endl;
    return 0;
}
```
但问题同样是超时，所以需要进行优化。
核心在于，如果直接处理，那么外循环是为了分别取每一个 yi 作为阈值，然后再嵌套一个内循环来计算当前阈值下预测正确的次数，这样就会直接达到 $O(n^2)$ 的复杂度。现在想要降低复杂度，可以考虑同样用一个循环来遍历每一个 yi （这是不能避免的），但是对于每一个指定的 y 即阈值，我可以直接给出其对应的预测正确次数。为了做到这一点，另外建立了两个 stu 类型的数组 a , b：

```cpp
stu *a = new stu[m]; // a[i].result 表示y小于a[i].y且result=0的数量
stu *b = new stu[m]; // b[i].result 表示y大于等于b[i].y且resuilt=1的数量
```
而建立 a , b 数组分别只需要 $O(n)$ 时间，加上排序 $O(nlgn)$ ，不会超过 $O(n^2)$ 。

```cpp
#include <iostream>
#include <algorithm>

using namespace std;

typedef struct{
    int y;
    int result;
}stu;

bool cmp(stu a, stu b){
    return a.y < b.y;
}

int threshold(stu *s, int m){
    int th; // 阈值
    int i;
    int num = 0; // 预测正确次数
    int maxnum = 0;
    int y,lasty=-1;
    stu *a = new stu[m]; // a[i].result 表示y小于a[i].y且result=0的数量
    stu *b = new stu[m]; // b[i].result 表示y大于等于b[i].y且resuilt=1的数量

    // 构建 a 数组
    for(i = 0; i < m; i++)
        a[i] = s[i];
    a[0].result = 0;
    for(i = 1; i < m; i++){
        /*
            单独看这里可能会有点错误，因为当连续出现相同的y时，
            也就是a[i].y==a[i-1].y时，二者的result可能不同，
            这是为了计算后面的result而做出的处理，并且只要在计算
            预测正确的次数时使用第一个y对应的result即可。
            （没什么理论依据，完全是照着例子推的）
        */
        if(a[i].y>=a[i-1].y && s[i-1].result==0)
            a[i].result = a[i-1].result+1;
        else
            a[i].result = a[i-1].result;
    }

    // 构建 b 数组
    for(i = 0; i < m; i++)
        b[i] = s[i];
    if(s[m-1].result==1)
        b[m-1].result = 1;
    else
        b[m-1].result = 0;
    for(i = m-2; i >= 0; i--){
        if(b[i].y<=b[i+1].y && s[i].result==1)
            b[i].result = b[i+1].result+1;
        else
            b[i].result = b[i+1].result;
    }

    for(i = 0; i < m; i++){
        if(s[i].y == lasty) // 相同y，只处理第一个
            continue;

        num = a[i].result + b[i].result;
        if(num > maxnum){
            maxnum = num;
            y = s[i].y;
        }
        if(num==maxnum && s[i].y>y)
            y = s[i].y;

        lasty = s[i].y;
    }

    th = y;
    return th;
}

int main(){
    int m, i, th;
    cin >> m;
    stu *s = new stu[m];
    for(i = 0; i < m; i++){
        cin >> s[i].y;
        cin >> s[i].result;
    }
    /*
        先按照y进行升序排序，使得在s[i]之前的y小于等于s[i].y，
        在s[i]之后的y大于等于s[i].y，从而能够方便的计算y
        小于s[i].y且result==0和y大于等于s[i].y且result==1
        即预测正确的次数。
    */
    stable_sort(s, s+m, cmp);
    th = threshold(s, m);
    cout << th << endl;
    return 0;
}
```

# 2020-09-1 称检测点查询
将一个检测点用一个结构体表示，其中含有 x 坐标，y 坐标，该检测点的编号，以及其到指定市民所在位置的距离。

然后依据该距离对结构体数组进行排序，输出前三位即可。

其中，如果会使用 C++ 中的 sort 函数则无需自己现场写一个排序。

```cpp
#include <iostream>
#include <algorithm>
using namespace std;

typedef struct {
    float x;
    float y;
    float d;
    int no;
} coordinate;

bool cmp(coordinate a, coordinate b){
    if(a.d == b.d)
        return a.no < b.no;
    return a.d < b.d;
}

void closest(coordinate *c, int n){
    for(int i = 0; i <= n; i++)
        c[i].d = sqrt(pow(c[0].x-c[i].x, 2)+pow(c[0].y-c[i].y, 2));
    sort(c, c+n+1, cmp);
    for(int i = 1; i <= 3; i++)
        cout << c[i].no << endl;

}
int main(){
    int n;
    cin >> n;
    coordinate *c = new coordinate[n+1];
    cin >> c[0].x;
    cin >> c[0].y;
    c[0].no = 0;
    for(int i = 1; i <= n; i++){
        cin >> c[i].x;
        cin >> c[i].y;
        c[i].no = i;
    }
    closest(c, n);
    return 0;
}
```

# 2020-09-2 风险人群筛查
n 个居民，每个居民 t 个时刻位置。
只需要一个两重循环来判断每个居民每个时刻是否处于高危区域，则不难得出经过高危区域的人数。其中计算逗留人数时，注意对每一个人，如果中途离开高危区域，则计数器应该置0，而后重新由0开始增加。

```cpp
#include <iostream>
using namespace std;

typedef struct {
    int *x;
    int *y;
}coordinate;


void howmany(int n, int k, int t, coordinate *c, int &pass, int &stay, int xl, int yd, int xr, int yu){
    int i, j;
    int ct = 0; // 计数连续出现在高危区域中的次数
    int yesorno = 0; // 表示该居民是否经过高危区域

    for(i = 0; i < n; i++){
        ct = 0;
        yesorno = 0;
        for(j = 0; j < t; j++){
            /*
                n 个居民，每个居民 t 个时刻，两重循环遍历。
                j 时刻处于高危区域，则 ct++ , yesorno++，
                否则 ct 置0。
                此外每次进入内循环先判断 ct 是否大于等于 k ，
                若成立，则满足逗留条件，无需再关心后续时刻位置。
            */
            if(ct >= k)
                break;
            if(c[i].x[j]>=xl&&c[i].x[j]<=xr&&c[i].y[j]>=yd&&c[i].y[j]<=yu){
                ct++;
                yesorno++;
            }
            else
                ct = 0;
        }
        if(yesorno != 0)
            pass++;
        if(ct >= k)
            stay++;
    }

}

int main(){
    int n,k,t,xl,yd,xr,yu;
    int i, j;
    int pass = 0; // 经过人数
    int stay = 0; // 逗留人数
    // 逗留包含经过
    cin >> n;
    cin >> k;
    cin >> t;
    cin >> xl;
    cin >> yd;
    cin >> xr;
    cin >> yu;

    // n 个结构体变量存储 n 个居民的信息
    coordinate *c = new coordinate[n];
    for(i = 0; i < n; i++){
        c[i].x = new int[t];
        c[i].y = new int[t];
    }

    /*
        每个结构体变量中有两个指针 x , y 分别指向两块
        连续空间，表示一个居民在各个时刻所处位置坐标
    */
    for(i = 0; i < n; i++){
        for(j = 0; j < t; j++){
            cin >> c[i].x[j];
            cin >> c[i].y[j];
        }
    }

    howmany(n, k, t, c, pass, stay, xl, yd, xr, yu);

    cout << pass << endl;
    cout << stay << endl;
    return 0;
}
```

# 2020-06-1 线性分类器
一共 n 个点，m 条直线。
对于每一条直线，首先将第一个点的横纵坐标代入直线方程，得到该点是位于直线上方还是下方（点的位置），同时记录点的类型。（初始条件）
随后按顺序对每一个点重复上述流程，每次将当前点的位置、类型与上一个点的位置、类型比较，如果都一致或都不一致，则保持正确，继续进行，否则错误，无法完成分类。

```cpp
#include <iostream>
using namespace std;

typedef struct {
    float x;
    float y;
    char type;
} point;

float gety(point p, int s0, int s1, int s2){
    float y;
    y = -(s0+s1*p.x)/s2;
    return y;
}

int main(){
    int n, m;
    int i, j;
    int up; // 表示该点是否处于直线上方
    int lup; // 表示前一个点是否处于直线上方
    float y; // 将该点 x 坐标代入直线方程求解 y
    char lastc; // 记录前一个点的类型
    int flag; // 表示是否正确分类
    cin >> n;
    cin >> m;

    point *p = new point[n];
    int *s0 = new int[m];
    int *s1 = new int[m];
    int *s2 = new int[m];

    for(i = 0; i < n; i++){
        cin >> p[i].x;
        cin >> p[i].y;
        cin >> p[i].type;
    }

    for(i = 0; i < m; i++){
        cin >> s0[i];
        cin >> s1[i];
        cin >> s2[i];
    }

    for(i = 0; i < m; i++){
        flag = 1;
        // 初始条件
        y = gety(p[0], s0[i], s1[i], s2[i]);
        if(p[0].y < y)
            up = 0;
        else
            up = 1;
        lastc = p[0].type;
        lup = up;

        for(j = 1; j < n; j++){
            y = gety(p[j], s0[i], s1[i], s2[i]);
            if(p[j].y < y)
                up = 0;
            else
                up = 1;
            /*
                如果第 j 个点与第 j-1 个点位置(不)一致、类型(不)一致则保持正确，继续判断。
                否则错误，无法完成分类。
            */
            if(up==lup && p[j].type==lastc);

            else if(up!=lup && p[j].type!=lastc);

            else{
                flag = 0;
                break;
            }
            lup = up;
            lastc = p[j].type;

        }

        if(flag)
            cout << "Yes" << endl;
        else
            cout << "No" << endl;
    }

    return 0;
}

```

# 2020-06-2 稀疏向量

```cpp
#include <iostream>

typedef struct {
    int index;
    int value;
} svector;

int main(){
    long long n, a, b;
    long long i, j;
    long long nj = 0;
    cin >> n;
    cin >> a;
    cin >> b;

    svector *u = new svector[a];
    svector *v = new svector[b];

    for(i = 0; i < a; i++){
        cin >> u[i].index;
        cin >> u[i].value;
    }
    for(i = 0; i < b; i++){
        cin >> v[i].index;
        cin >> v[i].value;
    }


    i = 0;
    j = 0;
    while(i<a&&j<b){
        if(u[i].index==v[j].index){
            nj = nj + u[i].value*v[j].value;
            i++;
            j++;
        }
        else if(u[i].index < v[j].index)
            i++;
        else
            j++;
    }

    cout << nj << endl;
    return 0;
}
```

# 2019-12-1 报数

唯一需要注意的点就是如何判断一个数中含有数字 7 。

其实也没别的办法，只能从低位到高位一个个判断。此时只要意识到对于一个整数 n ，n % 10 可以得到 n 的个位，同时 n / 10 可以去掉 n 的个位，那么本题就迎刃而解了。

```c++
#include <iostream>
using namespace std;

// 判断 n 中是否包含数字 7
bool is(int n){
    int t = 0;
    while(n!=0&&t!=7){
        t = n % 10; // t 是 n 的个位，循环相当于每次取最低位即判断每一位
        n = n / 10; // n 每次除以 10 即每次扔掉个位数
    }
    if(t==7)
        return true;
    else
        return false;
}
int main(){
    int n;
    cin >> n;
    int skip[4] = {0};
    int i = 0; // 表示当前数（包括跳过的）
    int c = 0; // 表示当前报数（不包括跳过的）

    while(c < n){
        i++; // 得到当前数
        // 判断 i 是否是 7 的倍数以及是否含有数字 7
        if(i%7==0||is(i)){
            skip[(i+3)%4] += 1;
            continue;
        }
        c++;
    }

    for(i = 0; i < 4; i++)
        cout << skip[i] << endl;

    return 0;
}
```
